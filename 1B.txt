import java.util.*;

public class Main {

    // Convert column number to Excel column name
    static String numberToColumn(int num) {
        StringBuilder sb = new StringBuilder();

        while (num > 0) {
            num--;
            sb.append((char) ('A' + (num % 26)));
            num /= 26;
        }

        return sb.reverse().toString();
    }

    // Convert Excel column name to column number
    static int columnToNumber(String col) {
        int num = 0;

        for (int i = 0; i < col.length(); i++) {
            num = num * 26 + (col.charAt(i) - 'A' + 1);
        }

        return num;
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int n = sc.nextInt();

        while (n-- > 0) {
            String s = sc.next();

            // Check if format is RXCY
            if (s.matches("R\\d+C\\d+")) {

                int cIndex = s.indexOf('C');
                int row = Integer.parseInt(s.substring(1, cIndex));
                int col = Integer.parseInt(s.substring(cIndex + 1));

                System.out.println(numberToColumn(col) + row);

            } else {

                int i = 0;
                while (Character.isLetter(s.charAt(i))) {
                    i++;
                }

                String col = s.substring(0, i);
                String row = s.substring(i);

                System.out.println("R" + row + "C" + columnToNumber(col));
            }
        }

        sc.close();
    }
}
